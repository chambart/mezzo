Module type Pprint.REPRESENTATION


module type REPRESENTATION = sig .. end
type representation
The type of value representation
val variant : Pprint.type_name ->
       Pprint.constructor ->
       Pprint.tag ->
       representation list ->
       representation
variant type_name data_constructor_name tag arguments Given information about the variant and its arguments, this function produces a new value representation.
val record : Pprint.type_name ->
       (Pprint.record_field * representation) list ->
       representation
record type_name fields Given a type name and a list of record fields, this function produces the value representation of a record.
val tuple : representation list ->
       representation
tuple arguments Given a list of value representation this function produces a new value representation.
