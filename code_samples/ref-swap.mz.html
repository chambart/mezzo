1 mutable data ref a = Ref { contents: a }

 3 val sswap [a, b]
 4   (consumes r: ref a, consumes s: ref b)
 5 : (| r @ ref b * s @ ref a)
 6 =
 7   let tmp = r.contents in
 8   r.contents <- s.contents;
 9   s.contents <- tmp

11 mutable data pool a = Pool {} adopts ref a

13 val wswap [a] (p: pool a, r: dynamic, s: dynamic) : () =

15     take r from p;
16     take s from p;
