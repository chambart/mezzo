 4 (* We don't have type abbreviations, so we have to wrap the type [a]
 5    in a one-field record. *)
 6 
 7 data id a =
 8   Id { value: a }
 9 
10 val return [a] (consumes x: a) : id a =
11   Id { value = x }
12 
13 val bind [a, b] (m: id a, k: a -> id b) : id b =
14   k m.value

16 (* The error monad. *)
17 
18 data option a =
19   Some { value: a }
20 | None
21 
22 val return [a] (consumes x: a) : option a =
23   Some { value = x }
24 
25 val bind [a, b] (m: option a, k: a -> option b) : option b =
26   match m with
27   | Some ->
28       k m.value
29   | None ->
30       None
31   end

33 (* The state-passing monad. *)
34 
35 data state s a =
36   State { computation: (consumes s) -> (s, a) }
37 
38 val return [s, a] duplicable a => (x: a) : state s a =
39   let c (consumes q: s) : (s, a) =
40     (q, x)
41   in
42   State { computation = c }
43 
44 val bind [s, a, b] (m: state s a, k: a -> state s b) : state s b =
45   let c (consumes q: s) : (s, b) =
46     let q, x = m.computation q in
47     (k x).computation q
48   in
49   State { computation = c }

51 (* The permission-passing monad. *)
52 
53 data ppm (p : perm) a =
54   PPM { computation: (| p) -> a }
55 
56 val return [p : perm, a] duplicable a => (x: a) : ppm p a =
57   (* We are forced to declare [a] as duplicable because the following
58      computation captures [a] in its closure. Should we use one-shot
59      functions? Or should we not use functions at all? Or is it just
60      a natural consequence of the fact that we have a non-indexed
61      state monad? *)
62   let c (| p) : a = x in
63   PPM { computation = c }
64 
65 val bind [p : perm, a, b] (m: ppm p a, k: a -> ppm p b) : ppm p b =
66   let c (| p) : b =
67     let x : a = m.computation() in
68     (k x).computation()
69   in
70   PPM { computation = c }
