 4 (* Data type definitions. *)
 5 
 6 mutable data cell a =
 7   Cell { value: a; next: dynamic }
 8 
 9 mutable data bag a =
10   Empty { head: (); tail: () }
11 | NonEmpty { head: dynamic; tail: dynamic }
12 adopts cell a
13 
14 (* The real functions. *)
15 
16 val create [a] (): bag a =
17   Empty { head = (); tail = () }
18 
19 val insert [a] (consumes x: a, b: bag a): () =
20   let c = Cell {
21     value = x; next = ()
22   } in
23   c.next <- c;
24   give c to b;
25   match b with
26   | Empty ->
27       tag of b <- NonEmpty;
28       b.head <- c;
29       b.tail <- c
30   | NonEmpty { tail } ->
31       take tail from b;
32       tail.next <- c;
33       give tail to b;
34       b.tail <- c
35   end
36 
37 val retrieve [a] (b: bag a): option a =
38   match b with
39   | Empty ->
40       none
41   | NonEmpty { head; tail } ->
42       take head from b;
43       if head == tail then begin
44         tag of b <- Empty;
45         b.head <- ();
46         b.tail <- ()
47       end else begin
48         b.head <- head.next
49       end;
50       let x = head.value in
51       some x
52   end
