1 open core
 2 (* This file implements a destination-passing style, almost tail-recursive
 3  * version of [List.map] for immutable lists. It uses a temporary mutable cell
 4  * that is later on "frozen" when it's ready. *)
 5 
 6 data list a = Cons { head: a; tail: list a } | Nil
 7 
 8 mutable data cell a = Cell { head: a; tail: () }
 9 
10 (* This version is tail-recursive. (The compiler may or may not be smart enough
11  * to figure out that "tag of x0 <- Cons" is a no-op, but this does not matter.) *)
12 val rec map1 [a, b] (
13           f: a -> b,
14           consumes c0: cell b,
15           consumes xs: list a
16         ): (| c0 @ list b)
17       =
18   match xs with
19   | Nil ->
20       (* c0 @ Cell { head: b; tail: () } *)
21       c0.tail <- Nil;
22       (* c0 @ Cell { head: b; tail: Nil } *)
23       tag of c0 <- Cons
24       (* c0 @ Cons { head: b; tail: Nil } *)
25       (* c0 @ list b *)
26   | Cons { head; tail } ->
27       (* c0 @ Cell { head: b; tail: () } *)
28       let c1 = Cell { head = f head; tail = () } in
29       (* c0 @ Cell { head: b; tail: () } * c1 @ Cell { head: b; tail: () } *)
30       c0.tail <- c1;
31       (* c0 @ Cell { head: b; tail = c1 } * c1 @ Cell { head: b; tail: () } *)
32       tag of c0 <- Cons;
33       (* c0 @ Cons { head: b; tail = c1 } * c1 @ Cell { head: b; tail: () } *)
34       map1 (f, c1, tail)
35       (* c0 @ Cons { head: b; tail = c1 } * c1 @ list b *)
36       (* c0 @ Cons { head: b; tail: list b } *)
37       (* c0 @ list b *)
38   end
39 
40 (* We need to unroll the recursion to take into account empty lists. *)
41 
42 val map [a, b] (f: a -> b, consumes xs: list a): list b =
43   match xs with
44   | Nil ->
45       xs
46   | Cons { head; tail } ->
47       let c = Cell { head = f head; tail = () } in
48       map1 (f, c, tail);
49       c
50   end
