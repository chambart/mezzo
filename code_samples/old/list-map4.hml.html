1 (* This file implements a destination-passing style, almost tail-recursive
 2  * version of [List.map] for immutable lists. It uses a temporary mutable cell
 3  * that is later on "freezed" when it's ready. *)
 4 
 5 data list a = Cons { head: a; tail: list a } | Nil
 6 mutable data xlist a = XCons { head: a; tail: list a } | XNil
 7 
 8 (* This version will be tail-rec if the compiler is smart enough to figure out
 9  * that "tag of x0 <- Cons" is a no-op. *)
10 val rec map1 [a, b] (
11           f: a -> b,
12           consumes c0: XCons { head: b; tail: () },
13           consumes xs: list a
14         ): (| c0 @ list b)
15       =
16   match xs with
17   | Nil ->
18       (* c0 @ XCons { head: b; tail: () } *)
19       c0.tail <- Nil;
20       (* c0 @ XCons { head: b; tail: Nil } *)
21       tag of c0 <- Cons
22       (* c0 @ Cons { head: b; tail: Nil } *)
23       (* c0 @ list b *)
24   | Cons { head; tail } ->
25       (* c0 @ XCons { head: b; tail: () } *)
26       let c1 = XCons { head = f head; tail = () } in
27       (* c0 @ XCons { head: b; tail: () } * c1 @ XCons { head: b; tail: () } *)
28       c0.tail <- c1;
29       (* c0 @ XCons { head: b; tail = c1 } * c1 @ XCons { head: b; tail: () } *)
30       map1 (f, c1, tail);
31       (* c0 @ XCons { head: b; tail = c1 } * c1 @ list b *)
32       tag of c0 <- Cons
33       (* c0 @ Cons { head: b; tail = c1 } * c1 @ list b *)
34       (* c0 @ list b *)
35   end
36 
37 
38 (* We need to unroll the recursion to take into account empty lists. *)
39 val map [a, b] (f: a -> b, consumes xs: list a): list b =
40   match xs with
41   | Nil ->
42       xs
43   | Cons { head; tail } ->
44       let c = XCons { head = f head; tail = () } in
45       map1 (f, c, tail);
46       c
47   end
