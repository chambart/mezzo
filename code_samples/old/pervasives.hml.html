16 (* The physical equality check (==) defined below returns, in case of success, a
17  * permission revealing that the two identifiers are equal. *)
18 
19 data phys_equal (x :: TERM) (y :: TERM) =
20   | Equals { | x = y }
21   | NotEquals

42 val none =
43   None

52

55 (* References *)
56 
57 val newref [a] (consumes x: a): ref a =
58   Ref { contents = x }
59 
60 val (!) [a] (consumes x: ref a): (a | x @ ref unknown) =
61   x.contents
62 
63 val (:=) [a, b] (consumes x: ref a, consumes c: b): ( | x @ ref b) =
64   x.contents <- c
