1 (* Data type definitions. *)
 2 
 3 mutable data cell a =
 4   Cell { contents: a; next: dynamic }
 5 
 6 mutable data bag a =
 7   Empty { head: (); tail: () }
 8 | NonEmpty { head: dynamic; tail: dynamic }
 9 adopts cell a
10 
11 (* Some dummy definitions. We don't include pervasives because that would make
12  * debugging harder (many more identifiers in scope). *)
13 
14 data bool = True | False
15 data option a = Some { value: a } | None
16 
17 val (==) [a, b] (x: a, y: b): bool = fail
18 
19 (* The real functions. *)
20 
21 val create [a] (): bag a =
22   Empty { head = (); tail = () }
23 
24 val insert [a] (consumes x: a, b: bag a): () =
25   let c = Cell {
26     contents = x; next = ()
27   } in
28   c.next <- c;
29   give c to b;
30   match b with
31   | Empty ->
32       tag of b <- NonEmpty;
33       b.head <- c;
34       b.tail <- c
35   | NonEmpty { tail } ->
36       take tail from b;
37       tail.next <- c;
38       give tail to b;
39       b.tail <- c
40   end
41 
42 val retrieve [a] (b: bag a): option a =
43   match b with
44   | Empty ->
45       None
46   | NonEmpty { head; tail } ->
47       take head from b;
48       if head == tail then begin
49         tag of b <- Empty;
50         b.head <- ();
51         b.tail <- ()
52       end else begin
53         b.head <- head.next
54       end;
55       let x = head.contents in
56       Some { value = x }
57   end
