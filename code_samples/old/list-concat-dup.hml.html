1 data list a = Cons { head: a; tail: list a } | Nil
 2 
 3 val rec concat [a] duplicable a => (xs: list a, ys: list a): list a =
 4   match xs with
 5   | Nil ->
 6       ys
 7   | Cons ->
 8       Cons {
 9         head = xs.head;
10         (* The polymorphic instantiation is mandatory. If we omit it, we use a
11          * flexible variable α to represent the polymorphic variable a in the
12          * recursive call. Because α is not yet determined, we don't have any
13          * hypotheses on it, and we can't satisfy the (duplicable α) constraint
14          * that appears in the recursive call, so we fail. We could be more
15          * clever, and perform the subtraction anyway and, if the subtraction
16          * instantiates α, check afterwards that the instantiation we picked
17          * does satisfy (duplicable α).
18          *
19          * 20121011: the trick above doesn't work, because when type-checking a
20          * function call, we first subtract "duplicable a => (=x, =y)" and then
21          * we subtract "x @ a ∗ y @ list a", so that doesn't help. *)
22         tail = concat [a] (xs.tail, ys)
23       }
24   end
