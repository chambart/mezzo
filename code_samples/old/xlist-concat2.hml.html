1 mutable data list a =
 2   | Cons { head: a; tail: list a }
 3   | Nil
 4 
 5 mutable data xlist a =
 6   | XCons { head: a; tail: xlist a }
 7   | XNil
 8 
 9 val rec concat1 [a] (
10    consumes dst: XCons { head: a; tail: () },
11    consumes xs: list a,
12    consumes ys: list a): (| dst @ list a)
13   =
14   match explain xs with
15   | Nil ->
16       dst.tail <- ys;
17       tag of dst <- Cons
18   | Cons { head; tail } ->
19       let dst' = XCons {
20         head = head; tail = ()
21       } in
22       dst.tail <- dst';
23       concat1 (dst', tail, ys);
24       tag of dst <- Cons
25   end
26 
27 val concat [a] (
28    consumes xs: list a,
29    consumes ys: list a): list a
30   =
31   match xs with
32   | Nil ->
33       ys
34   | Cons { head; tail } ->
35       let dst = XCons {
36         head = head; tail = ()
37       } in
38       concat1 (dst, tail, ys);
39       dst
40   end
