 3 data t = T
 4 data u = U { left: t; right: t }
 5 
 6 (* Un exemple encore plus compliqué, où il y a du partage d'un côté mais pas de
 7    l'autre. Il est quand même possible d'obtenir une bonne permission en sortie
 8    pour le résultat de ce if, car le type [t] est duplicable ce qui permet
 9    d'utiliser (à gauche) [x @ T] deux fois. On obtient
10 
11    ∃(? :: TERM). v3 @ U { left = x; right = ? } ∗ ? @ T ∗ x @ T
12 
13    À noter qu'on ne peut plus obtenir [v3 @ u] si jamais l'on déclare [t] comme
14    étant exclusif.
15 *)
16 val v3 =
17   let x = T in
18   if explain True then begin
19     U { left = x; right = x }
20   end else begin
21     let y = T in
22     U { left = x; right = y }
23   end
