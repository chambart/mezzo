 2 
 3 data list a = Cons { head: a; tail: list a } | Nil
 4 
 5 val rec map [a1, a2, b] (
 6     f: (consumes x: a1) -> (b | x @ a2),
 7     consumes xs: list a1): (list b | xs @ list a2) =
 8   match xs with
 9   | Nil ->
10       Nil
11   | Cons { head; tail } ->
12       (* Woohoo! Works without instantiating the polymorphic variables by hand! *)
13       Cons { head = f head; tail = map (f, tail) }
14   end
