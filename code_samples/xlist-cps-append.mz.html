1 open core
 2 mutable data xlist a =
 3   | XCons { head: a; tail: xlist a }
 4   | XNil
 5 
 6 val rec cps_mappend [a, b, p : perm] (
 7   consumes x: xlist a,
 8   consumes y: xlist a,
 9   k: (consumes xlist a | consumes p) -> b
10   | consumes p
11 ) : b =
12   match x with
13   | XNil ->
14       k y
15   | XCons ->
16       let f (consumes z : xlist a | consumes (x @ XCons { head: a; tail: unknown } * p)) : b =
17         x.tail <- z;
18         k x
19       in
20       cps_mappend (x.tail, y, f)
21   end
