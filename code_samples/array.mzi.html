 61 (* [consult] is a particular use of [get], with delimited scope, that does
 62    not require the element type to be duplicable. The function [f] does not
 63    have access to the array, so using [consult] instead of [get] is a way
 64    of ensuring that we only read one element at a time (and give it back). *)
 65 
 66 (* The type of [consult] advertises the fact that [consult] invokes [f]
 67    exactly once. Indeed, [f] is invoked at least once, otherwise [consult]
 68    would not be able to produce an element of type [b]. And [f] is invoked
 69    at most once, because [f] requires and consumes the permission [pre],
 70    which is not duplicable. *)
 71 
 72 (* A drawback of this approach is that the user has to work in what resembles
 73    CPS style. Could we somehow impose the same discipline and obtain the same
 74    effect in direct style? *)
 75 
 76 val consult : [a, pre : perm, b] (
 77   array a, int,
 78   f: (a | consumes pre) -> b
 79   | consumes pre
 80 ) -> b
 81 
 82 (* [update] is a particular combination of [get] and [set] that does not
 83    require the element type to be duplicable. *)
 84 
 85 (* The type of [update] advertises the fact that [update] invokes [f]
 86    exactly once. *)
 87 
 88 val update : [a, pre : perm, post : perm] (
 89   array a, int,
 90   f: (consumes a | consumes pre) -> (a | post)
 91   | consumes pre
