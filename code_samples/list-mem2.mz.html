 4 (* This is a variant of [list::mem]. It is defined in terms of [find] instead of
 5    being defined directly. *)
 6 
 7 val mem2 [a] duplicable a => (equal: (a, a) -> bool, x: a, consumes xs: list a) : list a =
 8   let p (y: a) : bool =
 9     equal (x, y)
10   in
11   find (p, xs)
