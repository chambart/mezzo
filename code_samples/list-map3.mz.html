 7 val rec map [a1, a2, b, p : perm] (
 8     f: (consumes x: a1 | p) -> (b | x @ a2),
 9     consumes xs: list a1 | p): (list b | xs @ list a2) =
10   match xs with
11   | Nil ->
12       Nil
13   | Cons { head; tail } ->
14       let v = f head in
15       Cons { head = v; tail = map (f, tail) }
16   end
