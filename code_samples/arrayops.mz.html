  7 (* Generic loops over a segment of an array. *)
  8 
  9 (* We use semi-open intervals of the form [i, j). That is, the index [i] is
 10    part of the interval, and the index [j] is not part of it. The interval
 11    is non-empty if and only if [i < j] holds. *)
 12 
 13 val rec fold_segment_up [a] (
 14   i: int, j: int, consumes accu: a,
 15   f: (int, consumes a) -> a
 16 ) : a =
 17   if i < j then
 18     fold_segment_up (i + 1, j, f (i, accu), f)
 19   else
 20     accu
 21 
 22 val rec fold_segment_down [a] (
 23   i: int, j: int, consumes accu: a,
 24   f: (int, consumes a) -> a
 25 ) : a =
 26   if i < j then
 27     fold_segment_down (i, j - 1, f (j - 1, accu), f)
 28   else
 29     accu
 30 
 31 val rec iter_segment_up [p : perm] (
 32   i: int, j: int,
 33   f: (int | p) -> ()
 34   | p
 35 ) : () =
 36   if i < j then begin
 37     f(i);
 38     iter_segment_up (i + 1, j, f)
 39   end
 40 
 41 val rec iter_segment_down [p : perm] (
 42   i: int, j: int,
 43   f: (int | p) -> ()
 44   | p
 45 ) : () =
 46   if i < j then begin
 47     f(j - 1);
 48     iter_segment_down (i, j - 1, f)
 49   end

 53 val fold_array_up [a, b] (
 54   x: array a, consumes seed: b,
 55   f: (int, consumes b | x @ array a) -> b
 56 ) : b =
 57   fold_segment_up [a = (b | x @ array a)] (0, length x, seed, f)
 58   (* WISH could we get rid of this type application? *)

 61   x: array a, consumes seed: b,
 62   f: (int, consumes b | x @ array a) -> b
 63 ) : b =
 64   fold_segment_down [a = (b | x @ array a)] (0, length x, seed, f)

 67 val iter_array_up [a, p : perm] (
 68   x: array a,
 69   f: (int | x @ array a * p) -> ()
 70   | p
 71 ) : () =
 72   iter_segment_up (0, length x, f)

 74 val iter_array_down [a, p : perm] (
 75   x: array a,
 76   f: (int | x @ array a * p) -> ()
 77   | p
 78 ) : () =
 79   iter_segment_down [ (x @ array a * p) ] (0, length x, f)

 81 (* A variant of the above loops, where [f] receives not only an index [i],

 84 val foldi_up [a, b] (
 85   x: array a, consumes seed: b,
 86   f: (int, a, consumes accu: b) -> b
 87 ) : b =
 88   fold_array_up (x, seed, fun (i: int, consumes accu: b | x @ array a) : b =
 89     consult (x, i, fun (v: a | consumes accu @ b) : b =
 90       f (i, v, accu)
 91     )
 92   )

196 val map [a, b, p : perm] (
197   x: array a,
198   f: (a | p) -> b
199   | p
200 ) : array b =
201   (* It seems to work when [p] is not there and [x @ array a] must be guessed, but not

216 (* Filling a segment with a value. *)
217 
218 val rec fill_segment [a] duplicable a => (x: array a, i: int, j: int, v: a) : () =
219   if i < j then begin
220     set (x, i, v);
221     fill_segment (x, i + 1, j, v)
222   end
223 
224 val fill [a] duplicable a => (x: array a, i: int, len: int, v: a) : () =
225   fill_segment (x, i, i + len, v)
226 
227 val fill_array [a] duplicable a => (x: array a, v: a) : () =
228   fill_segment (x, 0, length x, v)

248 (* Converting a list to an array. *)
249 
250 (* This less elegant and less compact version is defined directly. Due to its
251    use of [create], it is restricted to a duplicable element type. I keep only
252    as a test of the type-checker. *)
253 
254 val list2array [a] duplicable a => (xs: list::list a) : array a =
255   match xs with
256   | Nil ->
257       zero ()
258   | Cons { head; tail } ->
259       let x = create (1 + list::length tail, head) in
260       let rec fill (i: int, xs: list::list a | x @ array a) : =x =
261         match xs with
262         | Nil ->
263             x
264         | Cons ->
265             set (x, i, xs.head);
266             fill (i + 1, xs.tail)
267         end
268       in
269       fill (1, tail)
270   end
271 
272 (* This elegant and compact version is defined in terms of [init]. *)
273 
274 val list2array [a] (consumes xs: list::list a) : array a =
275   let n = list::length xs in
276   let r = newref xs in
277   init (n, fun (i: int | r @ ref (list::list a)) : a =
278     match !r with
279     | Cons { head; tail } ->
280         r := tail;
281         head
282     | Nil ->
283         fail (* impossible *)
284     end
285   )

293 (* Sorting. *)
294 
295 (* This is a port of OCaml's ternary heap sort. *)
296 
297 (* The three children of the node [i] are [i31], [i31+1], and [i31+2], where
298    [i31] stands for [i*3+1]. *)
299 
300 val sort [a, p : perm] duplicable a => (
301   cmp: (a, a | p) -> int,
302   x: array a
303   | p) : () =
304 
305   (* In all of the auxiliary functions that follow, [l] is the length of the
306      array [x]. *)
307 
308   (* The function [maxson] tells which of the children of the node [i] has
309      the greatest value. This node may have zero, one, two, or three children.
310      If there is at least one child, then the call returns the index of the
311      maximum child to the [normal] continuation. If there is zero child, then
312      the call returns the index [i] to the [bottom] continuation. This is a
313      double-barreled CPS encoding of the original OCaml code, which uses an
314      exception to signal that we have hit the bottom. *)
315 
316   (* The [consumes] annotation on the three occurrences of [x @ array a] below
317      seems to be optional. I believe that it should in principle be present.
318      By omitting this annotation, we would be committing to preserving this
319      permission, so we would be fixing a more specific type for [maxson]. In
320      essence, we would be specializing the answer type [b] to a type of the
321      form [b | x @ array a]. *)
322 
323   let maxson [b] (
324     l: int, i: int,
325     bottom: (int | consumes (p * x @ array a)) -> b,
326     normal: (int | consumes (p * x @ array a)) -> b
327                  | consumes (p * x @ array a))  : b =
328     let i31 = i+i+i+1 in
329     if i31+2 < l then begin
330       (* If there are three children, find which one has the maximum value. *)
331       let max = newref i31 in
332       if cmp (get (x, i31),  get (x, i31+1)) < 0 then max := i31+1;
333       if cmp (get (x, !max), get (x, i31+2)) < 0 then max := i31+2;
334       normal !max
335     end
336     else if i31+1 < l then begin
337       (* If there are just two children, find which one has the maximum value. *)
338       let max = if cmp (get (x, i31), get (x, i31+1)) < 0 then i31+1 else i31 in
339       normal max
340     end
341     else if i31 < l then
342       (* If there is just one child, return it. *)
343       normal i31
344     else
345       (* If there are no children, we have hit the bottom. Return [i]
346          via the exceptional continuation. *)
347       bottom i
348   in
349 
350   (* The function [trickledown] trickles the element [e] down node [i].
351      The normal continuation is specialized to an empty continuation. *)
352 
353   let rec trickledown (
354     l: int, i: int, e: a,
355     bottom: (int | p * x @ array a) -> ()
356     | p * x @ array a
357   ) : () =
358     maxson (* [b = (| p * x @ array a)] *) (l, i, bottom, fun (j: int | p * x @ array a) : () =
359       (* [j] is the maximum son of [i]. If the element [e] is ? than the
360          element found at [j], then send it down, otherwise write [e] into
361          node [i]. *)
362       if cmp (get (x, j), e) > 0 then begin
363         set (x, i, get (x, j));
364         trickledown (l, j, e, bottom)
365       end
366       else
367         set (x, i, e)
368     )
369   in
370 
371   (* The function [trickle] provides [trickledown] with a bottom continuation
372      that writes [e] into the childless node.  *)
373 
374   let trickle (l: int, i: int, e: a | p * x @ array a) : () =
375     trickledown (l, i, e, fun (i: int | p * x @ array a) : () =
376       set (x, i, e)
377     )
378   in
379 
380   (* The function [bubbledown] is analogous to [trickledown], but inserts a
381      a hole, rather than an element. It never terminates normally, i.e. it
382      always reaches a point where [maxson] invokes its bottom continuation.
383      For this reason, we can supply an identity continuation to [maxson]. *)
384 
385   let identity (i: int | p * x @ array a) : int =
386     i
387   in
388 
389   let rec bubbledown (
390     l: int, i: int
391     | p * x @ array a
392   ) : int =
393     maxson (* [b = (int | p * x @ array a)] *) (l, i, identity, fun (j: int | p * x @ array a) : int =
394       set (x, i, get (x, j));
395       bubbledown (l, j)
396     )
397   in
398 
399   (* Trickle up... *)
400 
401   let rec trickleup (i: int, e: a | p * x @ array a) : () =
402     let father = (i - 1) / 3 in
403     if cmp (get (x, father), e) < 0 then begin
404       set (x, i, get (x, father));
405       if father > 0 then trickleup (father, e) else set (x, 0, e)
406     end
407     else
408       set (x, i, e)
409   in
410 
411   (* The main algorithm. *)
412 
413   let l = length x in
414   iter_segment_down (0, (l + 1) / 3, fun (i: int | p * x @ array a) : () =
415     trickle (l, i, get (x, i))
416   );
417   iter_segment_down (2, l, fun (i: int | p * x @ array a) : () =
418     let e = get (x, i) in
419     set (x, i, get (x, 0));
420     trickleup (bubbledown (i, 0), e)
421   );
422   if l > 1 then
423     swap (x, 0, 1)
